답1) 내가 적은 답. 

#include<stdio.h>

int main(void) {
	int a[5][5] , n;

	n = 0;
	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 5; j++) {
			n++;
			a[i][j] = n;
		}
	}

	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 5; j++) {
			printf(" %d", a[i][j]);
		}
		printf("\n");
	}
}

답2) 책 답
#include<stdio.h>

int main(void) {
	int i, j, k;
	int a[5][5];

	k = 0;
	for (i = 0; i <= 4; i++) {
		for (j = 0; j <= 4; j++) {
			k++;
			a[i][j] = k;
		}
	}

	for (int x = 0; x <= 4; x++) {
		for (int y = 0; y <= 4; y++) {
			printf("%3d", a[x][y]);
		}
		printf("\n");
	}
}
