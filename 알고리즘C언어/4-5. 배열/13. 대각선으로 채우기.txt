답1) 내가 적은 답. 
#include<stdio.h>

int main(void) {
	int a[5][5] = { 0 };
	int n = 0 , s=-1 , e=0 , i ,j , x, y; 

	for (i = 0; i < 9; i++) {
		if (i < 5) {
			e++;
			x = -1;
			y = e;
		}
		else {
			e--;
			s++;
			x = s;
			y = 5;
		}
		for (j = 0; j < e; j++) {
			n++;
			x++;
			y--;
			a[x][y] = n;
		}
	}

	for (i = 0; i < 5; i++) {
		for (j = 0; j < 5; j++) {
			printf("%3d", a[i][j]);
		}
		printf("\n");
	}
}

답2) 책 답
#include<stdio.h>

int main(void) {
	int k, c, i, j, f, n;
	int a[5][5] = { 0 };

	k = 0;
	c = 1;
	i = 0;
	j = -1;

	f = 5;
	while (1) {
		for (n = 1; n <= f; n++) {
			k++;
			j += c;
			a[i][j] = k;
		}
		f--;
		if (f <= 0) {
			break;
		}
		for (n = 1; n <= f; n++) {
			k++;
			i += c;
			a[i][j] = k;
		}
		c *= -1;
	}
	for (int x = 0; x <= 4; x++) {
		for (int y = 0; y <= 4; y++) {
			printf("%3d", a[x][y]);
		}
		printf("\n");
	}
}
