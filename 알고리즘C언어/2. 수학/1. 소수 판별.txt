답1) 내가 적은 답. 

#include<stdio.h>

int main(void) 
{
	int i=1; int n;
	scanf_s("%d", &n);

	while (1) {
		i++;
		if (n%i == 0) {
			printf("%d는 소수가 아닙니다.", n);
			break;
		}
		if (i >= n-1) {
			printf("%d는 소수 입니다.", n);
			break;
		}
	}
}

#include<stdio.h>

int main(void)
{
	int a, i, j;
	scanf_s("%d", &a);
	i = a - 1;
	j = 2;

	while (1) 
	{
		if (j <= i) {
			if (a % j == 0) {
				printf("소수 아님");
				break;
			}
			else {
				j++;
			}
		}
		else {
			printf("소수");
			break;
		}

	}
}
