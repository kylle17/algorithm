문제)
  (77 x 1) + (76 x 2) + (75 x 3) + ... + (2 x 76) + (1 x 77)를 구하시오.

답1)
	
#include <stdio.h>

int main(void)
{
	int p = 0; int q = 0; int m = 0; int h = 0;

	do {
		p++;
		q = 78 - p;
		m = p * q;
		printf(" + %d*%d", q, p);
		h += m;
	} while (p < 77);
	printf(" = %d", h);

}
